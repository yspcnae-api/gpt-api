快速调用示例（Python）

# GPT-5 Chat 示例 (使用 gpt5-api 中转)
import requests

API_KEY = "your_gpt5_api_key"
API_URL = "https://api.gpt5-api.example/v1/chat/completions"

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

data = {
    "model": "gpt-5",
    "messages": [
        {"role": "system", "content": "你是一个专业的中文助理。"},
        {"role": "user", "content": "请简要说明 gpt5-api 中转站的优势。"}
    ],
    "max_tokens": 600
}

resp = requests.post(API_URL, headers=headers, json=data)
print(resp.json())
